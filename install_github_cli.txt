# Install GitHub CLI on Windows

# Option 1: Using winget (Windows Package Manager)
winget install --id GitHub.cli

# Option 2: Using Chocolatey (if you have it installed)
choco install gh

# Option 3: Using Scoop (if you have it installed)
scoop install gh

# After installation, authenticate and create repo:
gh auth login
gh repo create psx-ai-advisor --public --description "AI-powered stock market analysis tool for Pakistan Stock Exchange"
git remote add origin https://github.com/yourusername/psx-ai-advisor.git
git push -u origin main
